```tsx
// Fetch campaign with touches and related data
const { data: campaign, error: campaignError } = await supabase
  .from('campaigns')
  .select(`
    *,
    campaign_touches (
      id,
      touch_number,
      channel,
      delay_hours,
      template_id
    )
  `)
  .eq('id', params.id)
  .eq('business_id', business.id)
  .single()

// Fetch templates for preview - get all templates that are used by this campaign's touches
const templateIds = campaign?.campaign_touches
  ?.map((t: { template_id: string | null }) => t.template_id)
  .filter((id: string | null): id is string => id !== null) || []

const { data: templates } = templateIds.length > 0
  ? await supabase
      .from('message_templates')
      .select('id, name, subject, body, channel')
      .in('id', templateIds)
  : { data: [] }

// Create lookup map for quick access
const templateMap = new Map(
  (templates || []).map((t: MessageTemplate) => [t.id, t])
)
```

Then add preview section after form:

```tsx
// Add imports
import { PersonalizationPreview } from '@/components/ai/personalization-preview'
import { Tabs, TabsContent, TabsList, TabsTrigger } from '@/components/ui/tabs'

// In the edit page, add preview section after form:
// Show preview based on current campaign template

<div className="mt-8 space-y-4">
  <div className="flex items-center justify-between">
    <h3 className="font-medium">Message Preview</h3>
    <p className="text-sm text-muted-foreground">
      See how messages will appear to customers
    </p>
  </div>

  {campaign.campaign_touches?.length > 0 && (
    <Tabs defaultValue={`touch-1`}>
      <TabsList>
        {campaign.campaign_touches.map((touch) => (
          <TabsTrigger key={touch.id} value={`touch-${touch.touch_number}`}>
            Touch {touch.touch_number}
          </TabsTrigger>
        ))}
      </TabsList>

      {campaign.campaign_touches.map((touch) => {
        const template = touch.template_id ? templateMap.get(touch.template_id) : null
        return (
          <TabsContent key={touch.id} value={`touch-${touch.touch_number}`}>
            {template ? (
              <PersonalizationPreview
                templateBody={template.body}
                templateSubject={template.subject}
                channel={touch.channel}
                serviceType={campaign.service_type || undefined}
              />
            ) : (
              <div className="rounded-md border border-dashed p-8 text-center text-muted-foreground">
                No template selected for this touch
              </div>
            )}
          </TabsContent>
        )
      })}
    </Tabs>
  )}
</div>
```
