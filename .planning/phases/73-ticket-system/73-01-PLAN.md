```sql
CREATE TABLE public.project_tickets (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  project_id      UUID NOT NULL REFERENCES public.web_projects(id) ON DELETE CASCADE,
  business_id     UUID NOT NULL REFERENCES public.businesses(id) ON DELETE CASCADE,
  -- business_id denormalized for single-hop RLS (same pattern as campaign_enrollments)

  title           TEXT NOT NULL,
  description     TEXT,
  status          TEXT NOT NULL DEFAULT 'submitted'
    CHECK (status IN ('submitted', 'in_progress', 'completed')),
  priority        TEXT NOT NULL DEFAULT 'normal'
    CHECK (priority IN ('low', 'normal', 'high', 'urgent')),
  source          TEXT NOT NULL DEFAULT 'agency'
    CHECK (source IN ('agency', 'client_portal')),

  -- Overage tracking (TICK-05)
  is_overage      BOOLEAN NOT NULL DEFAULT false,
  overage_fee     NUMERIC(10,2),   -- populated when is_overage = true (e.g. 50.00)

  completed_at    TIMESTAMPTZ,
  internal_notes  TEXT,            -- agency-only, not shown to client

  created_at      TIMESTAMPTZ DEFAULT NOW(),
  updated_at      TIMESTAMPTZ DEFAULT NOW()
);

ALTER TABLE public.project_tickets ENABLE ROW LEVEL SECURITY;

CREATE POLICY "project_tickets_owner_all" ON public.project_tickets
  USING (
    business_id IN (SELECT id FROM public.businesses WHERE user_id = auth.uid())
  )
  WITH CHECK (
    business_id IN (SELECT id FROM public.businesses WHERE user_id = auth.uid())
  );

-- For client portal inserts: use service-role client in the Route Handler (Phase 74).
-- No anon INSERT policy needed — service-role bypasses RLS.

CREATE INDEX idx_project_tickets_project_id ON public.project_tickets(project_id);
CREATE INDEX idx_project_tickets_business_id ON public.project_tickets(business_id);
CREATE INDEX idx_project_tickets_status ON public.project_tickets(business_id, status, created_at DESC);
-- Partial index for "open tickets" — used in operator all-tickets view filters
CREATE INDEX idx_project_tickets_open
  ON public.project_tickets(project_id, created_at DESC)
  WHERE status IN ('submitted', 'in_progress');
```

**Atomic RPC for limit enforcement (TICK-03):**

This RPC is called from the operator "create ticket" action AND the client portal Route Handler (Phase 74). It serializes concurrent submissions by locking the web_projects row.

The RPC accepts:
- p_project_id UUID
- p_business_id UUID
- p_title TEXT
- p_description TEXT
- p_source TEXT ('agency' | 'client_portal')
- p_author_name TEXT (for the first message, nullable)
- p_body TEXT (first message body, nullable)
- p_monthly_limit INT (passed from app — REVISION_LIMITS constant)
- p_is_overage BOOLEAN (true when user confirmed $50 overage in UI)

Logic:
1. Lock the web_projects row: `SELECT id FROM web_projects WHERE id = p_project_id FOR UPDATE`
2. Count existing non-overage tickets in current calendar month:
   `SELECT COUNT(*) FROM project_tickets WHERE project_id = p_project_id AND is_overage = false AND created_at >= date_trunc('month', NOW())`
3. If count < p_monthly_limit OR p_is_overage = true: INSERT ticket, optionally INSERT first message, RETURN the new ticket id and status 'ok'
4. If count >= p_monthly_limit AND p_is_overage = false: RETURN status 'over_limit', count, limit (no insert)

Return type: TABLE(status TEXT, ticket_id UUID, current_count INT, monthly_limit INT)

```sql
CREATE OR REPLACE FUNCTION submit_ticket_with_limit_check(
  p_project_id    UUID,
  p_business_id   UUID,
  p_title         TEXT,
  p_description   TEXT,
  p_source        TEXT,
  p_author_name   TEXT,
  p_body          TEXT,
  p_monthly_limit INT,
  p_is_overage    BOOLEAN
)
RETURNS TABLE(status TEXT, ticket_id UUID, current_count INT, monthly_limit INT)
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_ticket_id UUID;
  v_count     INT;
BEGIN
  -- 1. Lock project row to serialize concurrent submissions
  PERFORM id FROM web_projects WHERE id = p_project_id FOR UPDATE;

  -- 2. Count non-overage tickets submitted this calendar month
  SELECT COUNT(*) INTO v_count
  FROM project_tickets
  WHERE project_id = p_project_id
    AND is_overage = false
    AND created_at >= date_trunc('month', NOW());

  -- 3. Enforce limit (unless this is an already-confirmed overage)
  IF v_count >= p_monthly_limit AND p_is_overage = false THEN
    RETURN QUERY SELECT
      'over_limit'::TEXT,
      NULL::UUID,
      v_count::INT,
      p_monthly_limit::INT;
    RETURN;
  END IF;

  -- 4. Insert ticket
  INSERT INTO project_tickets (
    project_id, business_id, title, description,
    status, source, is_overage, overage_fee
  )
  VALUES (
    p_project_id, p_business_id, p_title, p_description,
    'submitted', p_source,
    p_is_overage,
    CASE WHEN p_is_overage THEN 50.00 ELSE NULL END
  )
  RETURNING id INTO v_ticket_id;

  -- 5. Optionally insert first message (if body provided)
  IF p_body IS NOT NULL AND p_body != '' THEN
    INSERT INTO ticket_messages (
      ticket_id, business_id, author_type, author_name, body
    )
    VALUES (
      v_ticket_id, p_business_id,
      p_source, -- 'agency' or 'client_portal' maps to author_type
      p_author_name,
      p_body
    );
  END IF;

  RETURN QUERY SELECT
    'ok'::TEXT,
    v_ticket_id,
    (v_count + 1)::INT,
    p_monthly_limit::INT;
END;
$$;
```

IMPORTANT migration safety notes:
- All new columns are nullable or have static defaults — no NOT NULL without default on populated tables
- CHECK constraints on new tables (empty at migration time) — safe, no lock
- FOR UPDATE in the RPC serializes concurrent submissions without a separate lock table
- SECURITY DEFINER + SET search_path = public prevents search_path injection
