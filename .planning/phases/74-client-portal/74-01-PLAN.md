**Function 1: resolvePortalToken(token: string)**
- Queries `web_projects` table using service-role client
- `.select('id, business_id, domain, client_name, subscription_tier, status')` (select only what the portal displays)
- `.eq('portal_token', token).single()`
- Returns the row or null (do NOT throw — let the caller call notFound())
- Type the return as `{ id: string; business_id: string; domain: string | null; client_name: string | null; subscription_tier: string | null; status: string } | null`

**Function 2: getPortalQuota(projectId: string, tier: string | null)**
- Queries `project_tickets` table for tickets created in the current calendar month
- `.select('id', { count: 'exact', head: true })`
- `.eq('project_id', projectId)`
- `.gte('created_at', startOfCurrentMonth.toISOString())` where `startOfCurrentMonth` is computed as `new Date(new Date().getFullYear(), new Date().getMonth(), 1)`
- Plan limit: `tier === 'basic' ? 2 : tier === 'advanced' ? 4 : 2` (default 2 for unknown tiers — fail safe)
- Returns `{ used: number; limit: number; remaining: number }`

**Function 3: getPortalTickets(projectId: string)**
- Queries `project_tickets` joined with `ticket_messages`:
  ```
  .from('project_tickets')
  .select('id, title, description, status, source, created_at, updated_at, ticket_messages(id, author_type, author_name, body, attachment_urls, created_at)')
  .eq('project_id', projectId)
  .order('created_at', { ascending: false })
  .order('created_at', { ascending: true, referencedTable: 'ticket_messages' })
  ```
- Returns the array (empty array on error, never throws)

**app/portal/[token]/page.tsx** — Server Component, mirrors /complete/[token]/page.tsx exactly in structure:

```typescript
import { notFound } from 'next/navigation'
import type { Metadata } from 'next'
import { resolvePortalToken, getPortalQuota, getPortalTickets } from '@/lib/data/portal'
import { ClientPortal } from './client-portal'

interface Props { params: Promise<{ token: string }> }

export async function generateMetadata(): Promise<Metadata> {
  return { title: 'Client Portal', robots: 'noindex, nofollow' }
}

export default async function PortalPage({ params }: Props) {
  const { token } = await params

  // Resolve project via portal_token (service-role inside resolvePortalToken)
  const project = await resolvePortalToken(token)
  if (!project) notFound()

  // Fetch quota and tickets in parallel
  const [quota, tickets] = await Promise.all([
    getPortalQuota(project.id, project.subscription_tier),
    getPortalTickets(project.id),
  ])

  return (
    <div className="min-h-screen bg-gradient-to-b from-background to-muted/30">
      <ClientPortal
        token={token}
        project={project}
        quota={quota}
        initialTickets={tickets}
      />
    </div>
  )
}
```
