/**
 * Per-IP rate limit for the client portal ticket submission endpoint.
 * Allows 10 submissions per minute per IP to prevent spam while allowing
 * legitimate clients. Uses fixedWindow so the budget resets cleanly each minute.
 */
export const portalRatelimit = redis
  ? new Ratelimit({
      redis,
      limiter: Ratelimit.fixedWindow(10, '1 m'),
      prefix: 'ratelimit:portal',
    })
  : null

/**
 * Check rate limit for the portal ticket submission endpoint by IP address.
 * Bypasses in development if Upstash not configured.
 */
export async function checkPortalRateLimit(ip: string): Promise<{
  success: boolean
  remaining?: number
}> {
  if (!portalRatelimit) {
    return { success: true, remaining: 999 }
  }

  const result = await portalRatelimit.limit(ip)
  return {
    success: result.success,
    remaining: result.remaining,
  }
}

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { createServiceRoleClient } from '@/lib/supabase/service-role'
import { checkPortalRateLimit } from '@/lib/rate-limit'
import { portalTicketSchema } from '@/lib/validations/portal-ticket'

/**
 * POST /api/portal/tickets — Public portal ticket submission.
 *
 * No authentication required — project is resolved from the portal_token.
 * Uses service-role client to bypass RLS (same pattern as /api/complete, /api/intake).
 *
 * Flow:
 * 1. Rate limit by IP (10 submissions per minute)
 * 2. Validate request body against portalTicketSchema
 * 3. Resolve project + business from portal_token using service-role
 * 4. Enforce monthly revision quota (Basic: 2/mo, Advanced: 4/mo)
 * 5. Insert project_tickets row (source = 'client_portal')
 * 6. Return 201 on success
 */
export async function POST(req: NextRequest) {
  try {
    // Step 1: Rate limit by IP
    const ip = req.headers.get('x-forwarded-for')?.split(',')[0]?.trim() || 'unknown'
    const rateLimitResult = await checkPortalRateLimit(ip)
    if (!rateLimitResult.success) {
      return NextResponse.json(
        { error: 'Too many requests. Please wait before submitting again.' },
        { status: 429 }
      )
    }

    // Step 2: Validate body
    let body: unknown
    try {
      body = await req.json()
    } catch {
      return NextResponse.json({ error: 'Invalid JSON' }, { status: 400 })
    }

    const parsed = portalTicketSchema.safeParse(body)
    if (!parsed.success) {
      return NextResponse.json(
        { error: parsed.error.issues[0]?.message || 'Invalid input' },
        { status: 400 }
      )
    }

    const { token, title, description } = parsed.data

    // Step 3: Resolve project from portal_token (service-role — no RLS)
    const supabase = createServiceRoleClient()

    const { data: project } = await supabase
      .from('web_projects')
      .select('id, business_id, subscription_tier')
      .eq('portal_token', token)
      .single()

    if (!project) {
      return NextResponse.json(
        { error: 'Invalid or expired portal link' },
        { status: 404 }
      )
    }

    // Step 4: Enforce monthly quota
    const startOfMonth = new Date(new Date().getFullYear(), new Date().getMonth(), 1).toISOString()
    const { count: usedThisMonth } = await supabase
      .from('project_tickets')
      .select('id', { count: 'exact', head: true })
      .eq('project_id', project.id)
      .gte('created_at', startOfMonth)

    const planLimit = project.subscription_tier === 'advanced' ? 4 : 2
    const used = usedThisMonth ?? 0

    if (used >= planLimit) {
      return NextResponse.json(
        { error: `Monthly revision limit reached (${planLimit} per month). Please contact your agency for additional requests.` },
        { status: 429 }
      )
    }

    // Step 5: Insert ticket
    const { data: ticket, error: insertError } = await supabase
      .from('project_tickets')
      .insert({
        project_id: project.id,
        business_id: project.business_id,
        title,
        description: description || null,
        status: 'open',
        source: 'client_portal',
      })
      .select('id')
      .single()

    if (insertError || !ticket) {
      console.error('[portal/tickets] insert error:', insertError?.message)
      return NextResponse.json(
        { error: 'Failed to submit revision request. Please try again.' },
        { status: 500 }
      )
    }

    // Step 6: Return success
    return NextResponse.json({ success: true, ticketId: ticket.id }, { status: 201 })
  } catch (error) {
    console.error('[POST /api/portal/tickets] Internal error:', error instanceof Error ? error.message : 'Unknown error')
    return NextResponse.json({ error: 'Internal server error' }, { status: 500 })
  }
}
```
