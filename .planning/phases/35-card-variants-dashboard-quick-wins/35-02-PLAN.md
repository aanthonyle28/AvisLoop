    },
    success: {
      DEFAULT: "hsl(var(--success))",

      border: "hsl(var(--success-border))",

    },
    info: {
      DEFAULT: "hsl(var(--info))",
      bg: "hsl(var(--info-bg))",
      border: "hsl(var(--info-border))",
      foreground: "hsl(var(--info-foreground))",
    },
    error: {
      text: "hsl(var(--error-text))",
